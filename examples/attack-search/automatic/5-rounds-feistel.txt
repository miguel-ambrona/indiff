(* 5-rounds Feistel-Network automatic attack search *)

search_attack Indiff {

  rounds F1,F2,F3,F4,F5.

  oracle R (x0,x1) :=
    x2 = x0 + F1(x1);
    x3 = x1 + F2(x2);
    x4 = x2 + F3(x3);
    x5 = x3 + F4(x4);
    x6 = x4 + F5(x5);
    return x5, x6.

  oracle R^-1 (x5,x6) :=
    x4 = x6 + F5(x5);
    x3 = x5 + F4(x4);
    x2 = x4 + F3(x3);
    x1 = x3 + F2(x2);
    x0 = x2 + F1(x1);
    return x0, x1.

}.
